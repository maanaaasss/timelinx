## Installation

```bash
npm install @webpacked-timeline/core @webpacked-timeline/react @webpacked-timeline/ui
```

## Quick Start

```tsx
import { TimelineProvider } from '@webpacked-timeline/react';
import { DaVinciEditor } from '@webpacked-timeline/ui';
import { createEngine } from '@webpacked-timeline/core';
import '@webpacked-timeline/ui/styles/davinci';

const engine = createEngine({
  tracks: [],
  duration: 60000,
});

export default function App() {
  return (
    <TimelineProvider engine={engine}>
      <DaVinciEditor />
    </TimelineProvider>
  );
}
```
